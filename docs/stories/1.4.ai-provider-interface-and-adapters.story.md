```typescript
// src/providers/ai-provider.interface.ts
export interface AIProvider {
  readonly name: string;
  testConnection(): Promise<boolean>;
  generateText(prompt: string, options?: GenerateOptions): Promise<string>;
  streamText(prompt: string, options?: GenerateOptions): AsyncGenerator<string>;
}

export interface GenerateOptions {
  model?: string;
  maxTokens?: number;
  temperature?: number;
  systemPrompt?: string;
}

export class AIProviderError extends Error {
  constructor(
    message: string,
    public readonly provider: string,
    public readonly cause?: unknown,
  ) {
    super(message);
    this.name = 'AIProviderError';
  }
}
```

### Retry Logic Pattern (`BaseAIProvider`)

Exponential backoff with retryability detection via duck-typing (works across all three SDKs since OpenAI, Anthropic, and Gemini SDK errors all expose a `status` numeric property):

```typescript
protected async withRetry<T>(fn: () => Promise<T>, maxRetries = 3): Promise<T> {
  let lastError: unknown;
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (err) {
      lastError = err;
      if (!this.isRetryable(err) || attempt === maxRetries) throw err;
      const delay = Math.min(1000 * Math.pow(2, attempt), 10_000);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
  throw lastError;
}

private isRetryable(err: unknown): boolean {
  const status = (err as { status?: number }).status;
  return status === 429 || (typeof status === 'number' && status >= 500);
}
```

> **Rate limiting**: basic time-gap enforcement. `enforceRateLimit()` is called before each SDK call. Default `minIntervalMs = 0` (no enforced gap). The Epic 2 stories will configure actual limits per provider.

### Error Normalization Pattern

All providers must catch SDK-specific errors and re-throw as `AIProviderError`. Do NOT let raw SDK error types leak to callers:

```typescript
try {
  return await this.withRetry(() => /* SDK call */);
} catch (err) {
  if (err instanceof AIProviderError) throw err; // already normalized
  throw new AIProviderError(
    err instanceof Error ? err.message : 'Unknown error',
    this.name,
    err,
  );
}
```

### Mock Provider Design

`MockAIProvider` lives in `src/providers/` (not `tests/`) so it can be used both in tests and in any future `--dry-run` / `--mock` CLI flag:

```typescript
export class MockAIProvider implements AIProvider {
  readonly name = 'mock';
  constructor(private readonly _generateFn?: (prompt: string) => string) {}
  async testConnection(): Promise<boolean> { return true; }
  async generateText(prompt: string): Promise<string> {
    return this._generateFn ? this._generateFn(prompt) : `Mock response for: ${prompt}`;
  }
  async *streamText(prompt: string): AsyncGenerator<string> {
    yield this._generateFn ? this._generateFn(prompt) : `Mock stream for: ${prompt}`;
  }
}
```

### Code Quality Assessment

Solid, production-quality implementation. Architecture is clean — the interface/base/factory/provider/mock layering is textbook adapter pattern and keeps business logic decoupled from SDK specifics. Error normalisation via `AIProviderError` is applied consistently across all three providers, including the re-throw guard (`if (err instanceof AIProviderError) throw err`) which prevents double-wrapping. The `protected sleep()` override technique for test isolation is elegant: no fake timers, no brittle timing, no test framework leakage into source. The only class-level concern is the same module-level singleton deferred from Story 1.3 (factory export) — not applicable here since `AIProviderFactory.create()` is purely static.

### Refactoring Performed

- **File**: `tests/providers/anthropic-provider.test.ts`
  - **Change**: Added test `'passes systemPrompt as system field'` in the `generateText` describe block
  - **Why**: `OpenAIProvider` tests verified the `systemPrompt → system message` path; `AnthropicProvider` had no equivalent, leaving AC 5 partially untested. The Anthropic SDK uses a different shape (`system:` top-level field vs OpenAI's `messages[role:'system']`).
  - **How**: Mocks a successful response, calls `generateText` with `{ systemPrompt: '...' }`, asserts `mockMessagesCreate` received `expect.objectContaining({ system: '...' })`.

- **File**: `tests/providers/base-ai-provider.test.ts`
  - **Change**: Added test `'sleeps when called twice within the interval window'` in the `enforceRateLimit` describe block
  - **Why**: The two existing tests only covered the bypass paths (minIntervalMs = 0 and elapsed > interval). The active path — where sleep IS triggered — had no coverage, meaning a regression in the sleep-branch of `enforceRateLimit` would go undetected.
  - **How**: Uses `jest.spyOn` on the `TestProvider`'s `sleep` method (which is a no-op override); calls `enforceRateLimit` twice in quick succession, asserts `sleep` was called exactly once with a numeric argument.
