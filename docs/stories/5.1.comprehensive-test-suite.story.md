### Pre-existing EPERM Root Cause

`src/workflows/workspace-builder.ts` calls `Promise.all()` over all `WORKSPACE_DIRS` including `.cursor/rules/tmr`. The `FileSystemService.createDirectory` method wraps `fs.ensureDir` and re-throws any error as `FileSystemError`. In sandbox/restricted CI, creating `.cursor` directories under `/tmp` raises `EPERM`.

**Fix:** Replace `Promise.all` with `Promise.allSettled` and log each rejected directory as a warning via `logger.warn`. Continue despite individual EPERM failures — the workspace is still usable without the `.cursor` directory.

```typescript
// workspace-builder.ts — AFTER fix
import { logger } from '../utils/logger.js';
import { join } from 'node:path';
import { fileSystemService } from '../services/file-system.service.js';

export async function buildWorkspaceStructure(workspacePath: string): Promise<void> {
  const results = await Promise.allSettled(
    WORKSPACE_DIRS.map((dir) => fileSystemService.createDirectory(join(workspacePath, dir))),
  );
  results.forEach((result, i) => {
    if (result.status === 'rejected') {
      logger.warn(`Could not create directory ${WORKSPACE_DIRS[i]}: ${String(result.reason)}`);
    }
  });
}
```
