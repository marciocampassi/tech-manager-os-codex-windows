### Commander.js Integration Pattern

```typescript
import { Command } from 'commander';
import { createRequire } from 'module';

const require = createRequire(import.meta.url);
const pkg = require('../package.json') as { version: string };

export const program = new Command();

program
  .name('tm')
  .description('AI-powered CLI for engineering managers')
  .version(pkg.version, '-v, --version')
  .option('--verbose', 'enable verbose output')
  .option('--plain', 'disable colors and formatting')
  .option('--json', 'output as JSON');
```

> **Why `createRequire`?** ESM does not support `import json` by default without `--experimental-json-modules`. Using `createRequire` for JSON is the stable ESM-compatible pattern for Node 18+. `tsconfig.json` already has `"resolveJsonModule": true` for type checking purposes.

[Source: docs/prd/technical-assumptions.md#cli]

### Testing Pattern for Commander

Commander's `program.parseAsync()` requires `['node', 'tm', ...args]` as the argv array. To prevent `process.exit()` from killing tests, use `.exitOverride()`:

```typescript
program.exitOverride(); // throws instead of calling process.exit()

await expect(program.parseAsync(['node', 'tm', '--version']))
  .rejects.toThrow(); // Commander throws on --version with exitOverride
```

Alternatively, spy on `process.exit` with Jest:
```typescript
const exitSpy = jest.spyOn(process, 'exit').mockImplementation(() => { throw new Error('exit'); });
```
