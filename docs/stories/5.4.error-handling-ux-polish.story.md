4. Branded welcome message on `tmr init`
5. Help text is comprehensive and accurate for all commands including `tmr install` and `tmr update`
6. Accessibility flags function correctly (`--plain`, `--json`)

### Spinner Pattern

`ora` 8.x is the spinner library. In plain mode, ora must not render spinner frames. Use `ora({ isInteractive: !plain })` or check `plain` to no-op the spinner.

```typescript
// Expected display.ts signature
export interface SpinnerHandle {
  succeed(text?: string): void;
  fail(text?: string): void;
  update(text: string): void;
}

export function startSpinner(text: string, plain: boolean): SpinnerHandle {
  const start = Date.now();
  const elapsed = (): string => `(${((Date.now() - start) / 1000).toFixed(1)}s)`;
  if (plain) {
    process.stdout.write(`${text}...\n`);
    return {
      succeed: (t = text) => process.stdout.write(`${t} ${elapsed()}\n`),
      fail: (t = text) => process.stderr.write(`тЬЧ ${t} ${elapsed()}\n`),
      update: (t: string) => process.stdout.write(`${t}...\n`),
    };
  }
  const spinner = ora(text).start();
  return {
    succeed: (t = text) => { spinner.succeed(`${t} ${elapsed()}`); },
    fail: (t = text) => { spinner.fail(`${t} ${elapsed()}`); },
    update: (t: string) => { spinner.text = t; },
  };
}
```
